Can't loop:
  <p template="{type:attribute,         name:foo,         path:xxx.xxx.xxx}"></p>
  <p template="{type:textContent,                         path:xxx.xxx.xxx}"></p>
  <p template="{type:localizedContent , property:yyyy,    paths:[xxx.xxx.xxx]}"></p>

Can loop:
  <p template-loop="{selector:.foo, path:xxx.xxx.xxx, childSelector:xxx}"></p>

<template class=".foo">
  <p template="{type:textContent , arrayPath:xxx.xxx.xxx}"></p>
</template>
